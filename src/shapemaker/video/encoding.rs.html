53    pub fn encode(&mut self, output_file: impl Into<PathBuf>) -> Result<()> {
54        debug_time!("encode");
55
56        let output_file: PathBuf = output_file.into();
57
58        if output_file.exists() {
59            std::fs::remove_file(&output_file)?;
60        }
61
62        create_dir_all(
63            &output_file
64                .parent()
65                .expect("Given output file has no parent"),
66        )?;
67
68        let mut encoder = self.setup_encoder(&output_file)?;
69
70        self.progress_bar.set_position(0);
71        self.progress_bar.set_prefix("Rendering");
72        self.progress_bar.set_message("");
73
74        self.initial_canvas.load_fonts()?;
75        let initial_canvas = self.initial_canvas.clone();
76        let resolution = self.resolution;
77
78        let (tx, rx) = std::sync::mpsc::sync_channel::<(Duration, String)>(1_000);
79
80        let pb = self.progress_bar.clone();
81
82        let encoder_thread = thread::spawn(move || {
83            for (time, svg) in rx.iter() {
84                if svg.is_empty() {
85                    break;
86                }
87
88                encode_frame(
89                    &mut encoder,
90                    resolution,
91                    time,
92                    &initial_canvas,
93                    &svg,
94                )
95                .unwrap();
96
97                pb.set_position(time.as_millis() as _);
98                pb.set_message(milliseconds_to_timestamp(time.as_millis() as _));
99            }
100
101            encoder.stdin.take().unwrap().flush().unwrap();
102        });
103
104        self.render_all_frames(tx)?;
105
106        encoder_thread.join().expect("Encoder thread panicked");
107
108        Ok(())
109    }
